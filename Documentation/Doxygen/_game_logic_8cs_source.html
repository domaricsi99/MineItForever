132  public void MoveCharacter(Direction d, int mapID)

137  {

142  this.character.ChangeX(+7.5);
143  }
144  else if (d == Direction.Up)
145  {
146  this.character.ChangeY(-60);
147  }
148  }
149  else if (mapID == 1)
150  {
151  double moveSize = this.Movement(d);
152 
153  if (d == Direction.Left && this.character.Area.Left > 0)
154  {
155  this.character.ChangeX(-moveSize);
156  if (moveSize == 0 && !this.falling)
157  {
158  this.Mining(d);
159  }
160  }
161  else if (d == Direction.Right && this.character.Area.Right < Config.Width)
162  {
163  this.character.ChangeX(moveSize);
164  if (moveSize == 0 && !this.falling)
165  {
166  this.Mining(d);
167  }
168  }
169  else if (d == Direction.Up)
170  {
171  if (this.CanJumpMethod())
172  {
173  this.MapMovementDown();
174  }
175  }
176  else if (d == Direction.Down)
177  {
178  if (moveSize == 0 && !this.falling)
179  {
180  this.Mining(d);
181  }
182  else if (moveSize == 1)
183  {
184  this.MapMovementUpLadder();
185  }
186  }
187  else if (d == Direction.Climb)
188  {
189  if (moveSize == 1 && !this.falling)
190  {
191  this.MapMovementDownLadder();
192  }
193  }
194  }
195

340  public double Movement(Direction d)
341  {
342  Ore[,] renderedOres = this.MapPart();
343  double movementRange = 7.5;
344  int canPassThrough = 0;
345  Rect predictedChar = new Rect()
346  {
347  X = this.character.Area.X,
348  Y = this.character.Area.Y + 1,
349  Height = this.character.Area.Height - 2,
350  Width = this.character.Area.Width,
351  };
352 
353  switch (d)
354  {
355  case Direction.Left:
356  predictedChar.X -= 7.5;
357  foreach (var item in renderedOres)
358  {
359  if (item.Area.IntersectsWith(predictedChar) && item.OreType != "air" && item.OreType != "ladder" && item.OreType != "gate" && item.OreType != "gate2" && item.OreType != "gate3")
360  {
361  movementRange = (predictedChar.X + 7.5) - item.Area.Right - 1;
362  if (movementRange < 2)
363  {
364  return 0;
365  }
366  }
367  else if ((item.Area.IntersectsWith(predictedChar) && item.OreType == "gate")
368  || (item.Area.IntersectsWith(predictedChar) && item.OreType == "gate2")
369  || (item.Area.IntersectsWith(predictedChar) && item.OreType == "gate3"))
370  {
371  this.BackToMapOneScreen?.Invoke(this, EventArgs.Empty);
372  }
373  }
374 
375  break;
376  case Direction.Right:
377  predictedChar.X += 7.5;
378  foreach (var item in renderedOres)
379  {
380  if (item.Area.IntersectsWith(predictedChar) && item.OreType != "air" && item.OreType != "ladder" && item.OreType != "gate")
381  {
382  movementRange = item.Area.Left - 1 - (predictedChar.X + Config.MinerWidth - 7.5);
383  if (movementRange < 2)
384  {
385  return 0;
386  }
387  }
388  else if (item.Area.IntersectsWith(predictedChar) && item.OreType == "gate")
389  {
390  this.BackToMapOneScreen?.Invoke(this, EventArgs.Empty);
391  }
392  }
393 
394  break;
395  case Direction.Down:
396  predictedChar.Y += Config.MinerHeight;
397  foreach (var item in renderedOres)
398  {
399  if ((item.OreType == "ladder" || item.OreType == "air")
400  && item.Area.Left <= predictedChar.Left
401  && item.Area.Right >= predictedChar.Right
402  && item.Area.IntersectsWith(predictedChar))
403  {
404  canPassThrough = 1;
405  return canPassThrough;
406  }
407  else
408  {
409  canPassThrough = 0;
410  }
411  }
412 
413  return canPassThrough;
414  case Direction.Climb:
415  predictedChar.Y -= Config.MinerHeight;
416  foreach (var item in renderedOres)
417  {
418  if ((item.OreType == "ladder" || item.OreType == "air")
419  && item.Area.Left <= predictedChar.Left
420  && item.Area.Right >= predictedChar.Right
421  && item.Area.IntersectsWith(predictedChar))
422  {
423  canPassThrough = 1;
424  return canPassThrough;
425  }
426  else
427  {
428  canPassThrough = 0;
429  }
430  }
431 
432  return canPassThrough;
433  default:
434  break;
435  }
436 
437  return movementRange;
438  }

489  public Ore[,] MapPart()
490  {
491  Ore[,] renderedOres = new Ore[5, 5];
492  int intersectOreX = 2;
493  int intersectOreY = 2;
494 
495  for (int i = 0; i < this.ore.GetLength(0); i++)
496  {
497  for (int j = 0; j < this.ore.GetLength(1); j++)
498  {
499  if (this.character.Area.IntersectsWith(this.ore[i, j].Area))
500  {
501  intersectOreX = i;
502  intersectOreY = j;
503  }
504  }
505  }
506 
507  if (intersectOreX - 2 < 0)
508  {
509  intersectOreX = 2;
510  }
511 
512  if (intersectOreY - 2 < 0)
513  {
514  intersectOreY = 2;
515  }
516 
517  if ((intersectOreX + 2) >= this.ore.GetLength(0))
518  {
519  intersectOreX = this.ore.GetLength(0) - 3;
520  }
521 
522  if ((intersectOreY + 2) * 45 >= Config.Width)
523  {
524  intersectOreY = this.ore.GetLength(1) - 3;
525  }
526 
527  int startingPosY = intersectOreY;
528  for (int i = 0; i < renderedOres.GetLength(0); i++)
529  {
530  for (int j = 0; j < renderedOres.GetLength(1); j++)
531  {
532  renderedOres[i, j] = this.ore[intersectOreX - 2, intersectOreY - 2];
533  intersectOreY++;
534  }
535 
536  intersectOreY = startingPosY;
537  intersectOreX++;
538  }
539 
540  return renderedOres;
541  }
